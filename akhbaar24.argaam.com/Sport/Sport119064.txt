رحيمي ... خلافات الاتحادين حولت الفريق الى دار عجزة 

 أكــد الـدكـتـور مـدنـي رحيمي عـضـو شــرف نـــادي الاتــحــاد، أن الــخــلافــات بـــين رئــيــس مـجـلـس الإدارة الـحـالـي إبـراهـيـم الـبـلـوي وشقيقه منصور البلوي المشرف على الكرة، هي السبب الأول خلف تراجع أداء الفريق وتدهور أوضاع النادي على وجه العموم. وقـــــــال مـــدنـــي فــــي حــديــث لـ«الشرق الأوسط»: «أول الخلافات بين الرئيس وشقيقه تسببت في إلغاء عقد المدرب السابق الروماني بيتوركا، إذ كان إبراهيم وشقيقة ســلــطــان يــرغــبــان بــبــقــائــه، فـيـمـا عـارض منصور الأمـر ليتم إلغاء عـــقـــده، أمــــام آخــــر الاخــتــلافــات بينهما فتدور حول المغربي مبارك بـوصـوفـة، إذ يصر منصور على التعاقد معه لكن إبراهيم يرفض بدعوى قيمة عقده الكبيرة، وهذا مــا دفــع الأخــيــر إلــى تــرك الـنـادي والسفر إلى خارج السعودية حتى لا يقع في حرج ويجبر على توقيع العقد». وتابع: «بالنسبة للمشكلات الفنية فهي كثيرة في الفريق الأول، فأنا أعتبر فريق كـرة القدم فريق دار عجزة، ولو نظرنا إلى الفريق فــنــيــا تــجــد أهـــم خــط وهـــو خـط الوسط يلعب فيه الروماني سان مارتن (٣٦ عاما) ومحمد نور (٣٥ عاما) والغاني مونتاري (٣٢ عاما)وأعتقد أن هذا اللاعب كان عاطلا وبالنظر إلى عطائه في الملعب فهو محدود، فهل من المعقول أن يؤدي هذا الخط (خط الوسط) وجباته كاملة في هذا الوضع؟!». وواصل: «خط الوسط يحتاج إلى لاعبين نشطين لا أولئك الذين (يلعبون على الـواقـف)، ومـبـاراة الفيصلي كشفتهم فنيا وعرتهم خـــصـــوصـــا مــحــمــد نـــــور الــــذي كــان سـبـب الـهـزيـمـة، فـتـأخـره عن الالتحام مع صاحب الهدف الثاني يـؤكـد أنــه (نــور) لـم يـعـد لـديـه مايقدمه». وتــابــع رحـيـمـي فــي حـديـثـه: «أمـام المدافع حمد المنتشري فهو ليس من ضمن العجزة في الفريقٍ (بل منته كليا) ولا يملك مقومات اللاعب، أنا أضع علامة استفهام أمــــام إصـــــرار الإدارة والــجــهــاز الفني على إشراكه في كل مباراة واستبعاد اللاعب أحمد عسيري الــذي كــان سـيـشـكـل ثـنـائـيـا قـويـا مــع يـاسـين حـمـزة الــذي أثـبـت أن بـاسـتـطـاعـتـه الـلـعـب فــي مـركـزي الــظــهــيــر الأيــمــن والأيـــســـر وفــي مـتـوسـط الــدفــاع ولـقـد شـاهـدنـاه مع المنتخب السعودي في متوسط الــدفــاع الأيــســر بـيـنـمـا يـلـعـب في الاتحاد كظهير أيمن». وجــزم رحـيـمـي بــأن الـربـاعـي الـشـاب فـهـد المـولـد وعـبـد الـفـتـاح عسيري وجمال با جندوح وأحمد عـسـيـري مـحـاربـون لـسـبـب واحـد وهــو رفـضـهـم أن يـتـولـى سلطان البلوي إدارة أعمالهم الأمـر الذي جعلهم غير مرغوب بهم، ولاحظنا أن المولد هو الوحيد الذي يشارك في المباريات كونه لا يوجد بديل قد يحل مكانه في مركزه، ولو وجد هـذا البديل «حتى لو كـان سيئا» سيكون مصيره مقاعد البدلاء. وقال عضو الشرف الاتحادي إن تـعـاقـد الــنــادي مـع الأسـتـرالـي جيمس تـروسـي لـم يكن مـن أجل الفريق بـل مـن أجـل الضغط على  عبد الفتاح عسيري كي يوقع مع سلطان البلوي. وأشار رحيمي إلى أن التعاقد مـــع بــولــونــي كـــمـــدرب بــمــثــابــة«الكارثة الفنية» بحد ذاتها. وقـــــال: «بــولــونــي مــســيــر لا مخير، لقد أقام الاتحاد معسكر في إيطاليا وظـهـر لاعـبـوه بمستوى عـــادي جـــدا، ثــم عـسـكـر فــي جبل علي بدبي وقدم أسوأ مستوى له أمـام الفيصلي، وأنـا أتـسـاءل عن الـــدور الــذي يـقـوم بــه المـــدرب من ناحية تطوير مستوى فريق الكرة وأعتقد أن أمامه فرصة لتصحيح أوضـــاع الـفـريـق رغــم شخصيته الضعيفة». وأردف رحـيـمـي فـي حـديـثـه:«هــذا المــدرب فضح نفسه بنفسه وأكــد أن هـنـاك تـدخـلات فنية في شؤونه، فبعد مباراة الفيصلي أكد أن التشكيلة كانت خاطئة كما أن الإدارة صرحت بأن المدرب لا يريد جـمـال بـاجـنـدوح فيكيف يضعه في الاحتياط؟!» وواصل: «أما في شأن حامد البلوي المدير التنفيذي لفريق الكرة فقد أثبت أنه لا يملك أي من مقومات المدير التنفيذي ولا يفقه شيئا فـي عالم الـكـرة بدليل الوضع الفني للفريق، لذا أعتقد أن لديه مهام أخرى لا نعرفها وأتمنى أن يبتعد عن مجال كرة القدم لأنه ليس مجاله». ووصـــــف رحــيــمــي الإعــــلام الاتحادي بأنه «مسير لا مخير»، «بدليل غياب الانـتـقـادات الجادة لـلإدارة والمـدرب واللاعبين، وكأن الاتحاد هو المتصدر». وتابع رحيمي: «هـذا الإعـلام الــذي يـقـوده أحـد أعـضـاء الشرف ويوجهه كما يريد أضر بالاتحاد كثيرا، فأعضاء الشرف المؤثرون والـــداعـــمـــون لــلــنــادي ابــتــعــدوا بسبب التهجم من بعض الكاتب الاتـــحـــاديـــين عــلــيــهــم ووصــفــهــم بعبارات جارحة لا تليق بمراكزهم ومكانتهم في المجتمع فأنا أعتقد أن هـذا الإعـلام لا بد أن يقف عند حده».
