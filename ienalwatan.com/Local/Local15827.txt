ﺟﺎﻣﻌﺔ ﺍﻟﻤﺆﺳﺲ ﺗﻄﻠﻖ 7 ﺑﺮﺍﻣﺞ ﺿﻤﻦ ﻣﺒﺎﺩﺭﺓ ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ ﺑﺤﻲ ﺍﻟﺠﺎﻣﻌﺔ 

أﻃﻠﻖ ﻣﻌﺎﻟﻲ ﻣﺪﻳﺮ ﺍﻟﺠﺎﻣﻌﺔ ﺍﻷ‌ﺳﺘﺎﺫ ﺍﻟﺪﻛﺘﻮﺭ ﻋﺒﺪﺍﻟﺮﺣﻤﻦ ﺑﻦ ﻋﺒﻴﺪ ﺍﻟﻴﻮﺑﻲ ﻣﺒﺎﺩﺭﺓ ﺍﻟﺠﺎﻣﻌﺔ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﺿﻤﻦ ﻣﻠﺘﻘﻰ ﻣﻜﺔ ﺍﻟﺜﻘﺎﻓﻲ ﺗﺤﺖ ﺷﻌﺎﺭ “ﻛﻴﻒ ﻧﻜﻮﻥ ﻗﺪﻭﺓ” ﻭﺗﺴﺘﻬﺪﻑ ﺍﻟﻤﺒﺎﺩﺭﺓ ﺣﻲ ﺍﻟﺠﺎﻣﻌﺔ، ﻭﺗﺸﻤﻞ 7 ﺑﺮﺍﻣﺞ ﺗﻨﻤﻮﻳﺔ ﻭﺻﺤﻴﺔ  ﻭﻋﻠﻤﻴﺔ ﻭﻣﻌﺮﻓﻴﺔ  ﻭﺫﻟﻚ ﺑﺄﺭﺽ ﺍﻟﺨﺪﻣﺎﺕ ﺷﺮﻕ ﺷﺎﺭﻉ ﺍﻟﺠﺎﻣﻌﺔ ﺑﺤﻲ ﺍﻟﺠﺎﻣﻌﺔ، ﻭﺑﺤﻀﻮﺭ ﻣﻤﺜﻠﻴﻦ ﻟﻠﻘﻄﺎﻋﺎﺕ ﺍﻟﺤﻜﻮﻣﻴﺔ ﺍﻟﻤﺸﺎﺭﻛﺔ ﻓﻲ ﺍﻟﻤﺒﺎﺩﺭﺓ، ﻭﻋﻤﺪﺓ ﻭﺃﻋﻴﺎﻥ ﻭﺃﻫﺎﻟﻲ ﺣﻲ ﺍﻟﺠﺎﻣﻌﺔ.
  
 ﻭﺑﺪﺃ ﺍﻟﺤﻔﻞ ﺑﺂﻳﺎﺕ ﻣﻦ ﺍﻟﺬﻛﺮ ﺍﻟﺤﻜﻴﻢ ﺛﻢ ﺍﻟﻘﻰ ﺍﻻ‌ﺳﺘﺎﺫ ﻋﻼ‌ﺀ ﺑﻠﻪ ﺍﻟﻬﻠﺒﺎﻭﻱ ﻛﻠﻤﺔ ﺍﻷ‌ﻫﺎﻟﻲ ﻭﺍﻟﺘﻲ ﺷﻜﺮ ﻓﻴﻬﺎ ﺟﺎﻣﻌﺔ ﺍﻟﻤﻠﻚ ﻋﺒﺪﺍﻟﻌﺰﻳﺰ ﻋﻠﻰ ﺍﻫﺘﻤﺎﻣﻬﺎ ﺑﺤﻲ ﺍﻟﺠﺎﻣﻌﺔ ﻣﻦ ﺧﻼ‌ﻝ ﻣﺒﺎﺩﺭﺗﻬﺎ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﻭﺍﻟﺘﻲ ﺗﺄﺗﻲ ﺗﺮﺟﻤﺔ ﺣﻘﻴﻘﻴﺔ ﻟﺪﻭﺭ ﺍﻟﺠﺎﻣﻌﺔ ﺗﺠﺎﻩ ﺍﻟﻤﺠﺘﻤﻊ ﻭﺗﻨﻔﻴﺬ ﻣﻴﺪﺍﻧﻲ ﻟﺸﻌﺎﺭ ﻣﻠﺘﻘﻰ ﻣﻜﺔ ﺍﻟﺜﻘﺎﻓﻲ “ﻛﻴﻒ ﻧﻜﻮﻥ ﻗﺪﻭﺓ”، ﻋﻘﺐ ﺫﻟﻚ ﺷﺎﻫﺪ ﺍﻟﺤﻀﻮﺭ ﻋﺮﺿﺎً ﻣﺮﺋﻴﺎً ﻋﻦ ﺍﻟﻤﺒﺎﺩﺭﺓ ﻭﺃﻫﺪﺍﻓﻬﺎ، ﻭﻗﺪﻡ ﻧﺎﺩﻱ ﺍﻟﻤﺴﺮﺡ ﻣﺴﺮﺣﻴﺔ ﺗﻨﺎﻭﻟﺖ ﺃﻫﻤﻴﺔ ﺍﻟﻘﺪﻭﺓ ﻓﻲ ﺍﻟﻤﺠﺘﻤﻊ ﺇﺧﺮﺍﺝ ﻓﻴﺼﻞ ﺍﻟﺤﺮﺑﻲ، ﻭﻗﺪﻡ ﻧﺎﺩﻱ ﺍﻟﺘﺮﺍﺙ ﺍﻟﺸﻌﺒﻲ ﺑﻌﺮﻭﺽ ﺷﻌﺒﻴﺔ ﺷﺎﺭﻙ ﻓﻴﻬﺎ ﺃﻫﺎﻟﻲ ﺍﻟﺤﻲ.
  
 ﻭﺃﻋﻠﻦ ﻣﻌﺎﻟﻲ ﻣﺪﻳﺮ ﺍﻟﺠﺎﻣﻌﺔ ﺍﻷ‌ﺳﺘﺎﺫ ﺍﻟﺪﻛﺘﻮﺭ ﻋﺒﺪﺍﻟﺮﺣﻤﻦ ﺑﻦ ﻋﺒﻴﺪ ﺍﻟﻴﻮﺑﻲ ﺧﻼ‌ﻝ ﺍﻟﻜﻠﻤﺔ ﺍﻟﺘﻲ ﺍﻟﻘﻬﺎ ﻓﻲ ﺣﻔﻞ ﺍﻟﺘﺪﺷﻴﻦ ﻋﻦ ﻓﺘﺢ ﺻﺎﻻ‌ﺕ ﺍﻷ‌ﻧﺸﻄﺔ ﺍﻟﻤﺮﺍﻓﻖ ﺍﻟﺮﻳﺎﺿﻴﺔ ﻷ‌ﺑﻨﺎﺀ ﺣﻲ ﺍﻟﺠﺎﻣﻌﺔ ﻭﺗﺴﺨﻴﺮ ﻛﺎﻓﺔ ﺍﻹ‌ﻣﻜﺎﻧﺎﺕ ﻟﻬﻢ ﻟﻤﻤﺎﺭﺳﺔ ﻫﻮﻳﺎﺗﻬﻢ ﻭﻣﻮﺍﻫﺒﻬﻢ ﻣﻦ ﺧﻼ‌ﻝ ﺑﺮﺍﻣﺞ ﻭﺃﻧﺸﻄﺔ ﺳﺘﺴﻬﻢ ﻓﻲ ﺗﻄﻮﻳﺮ ﻗﺪﺭﺍﺗﻬﻢ ﺍﻟﻔﻜﺮﻳﺔ ﻭﺍﻟﻤﻬﺎﺭﻳﺔ، ﻭﺃﻛﺪ ﻣﻌﺎﻟﻴﻪ ﺃﻥ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﻭﺑﻌﺪ ﺗﻄﺒﻴﻖ ﺍﻟﺘﺠﺮﺑﺔ ﻋﻠﻰ ﺣﻲ ﺍﻟﺠﺎﻣﻌﺔ ﺳﻴﺘﻢ ﻓﻲ ﺑﺈﺫﻥ ﺍﻟﻠﻪ ﺗﻌﺎﻟﻰ ﺗﻨﻔﻴﺬﻩ ﻓﻲ ﺃﺣﻴﺎﺀ ﺃﺧﺮﻯ ﻓﻲ ﻣﺪﻳﻨﺔ ﺟﺪﺓ.
  
 ﻭﻗﺎﻝ ﻣﻌﺎﻟﻴﻪ “ﻣﻦ ﺛﻤﺎﺭ ﻣﺸﺮﻭﻉ “ﻛﻴﻒ ﻧﻜﻮﻥ ﻗﺪﻭﺓ” ﺃﻥ ﺑﺎﺩﺭﺕ ﺟﺎﻣﻌﺔ ﺍﻟﻤﻠﻚ ﻋﺒﺪﺍﻟﻌﺰﻳﺰ ﺑﻤﺒﺎﺩﺭﺓ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﺍﻟﺘﻲ ﻭﺍﻓﻖ ﻋﻠﻴﻬﺎ ﺻﺎﺣﺐ ﺍﻟﺴﻤﻮ ﺍﻟﻤﻠﻜﻲ ﺍﻷ‌ﻣﻴﺮ ﺧﺎﻟﺪ ﺍﻟﻔﻴﺼﻞ ﻭﺫﻟﻚ ﻣﻦ ﻣﻨﻄﻠﻖ ﺃﻥ ﻣﻔﻬﻮﻡ ﺍﻟﻘﺪﻭﺓ ﻻ‌ ﻳﻤﻜﻦ ﺃﻥ ﻳﺘﺄﺗﻰ ﺑﺄﺑﻌﺎﺩﻩ ﺟﻤﻴﻌﺎً ﺇﻻ‌ ﻣﻦ ﺧﻼ‌ﻝ ﺳﻠﺴﻠﺔ ﻣﺘﺪﺍﺧﻠﺔ ﺍﻟﻤﺤﺎﻭﺭ ﻭﺍﻷ‌ﺑﻌﺎﺩ،  ﻭﺟﺎﻣﻌﺔ “ﺍﻟﻤﺆﺳﺲ” ﺇﺫ ﺗﺘﺒﻨﻰ ﻓﻜﺮﺓ ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ ﻭﻛﻠﻬﺎ ﺛﻘﺔ ﻣﻦ ﺃﻥ ﺍﻟﻤﺒﺎﺩﺭﺓ ﺳﺘﺆﺩﻱ ﺇﻟﻰ ﺗﻌﺰﻳﺰ ﺟﻬﻮﺩ ﺍﻟﺘﻄﻮﻳﺮ ﻭﺍﻟﺘﻐﻴﻴﺮ ﺍﻹ‌ﻳﺠﺎﺑﻲ، ﻭﺗﺠﻮﻳﺪ ﺍﻷ‌ﺩﺍﺀ ﻓﻲ ﻛﻞ ﺍﻟﻤﺮﺍﻓﻖ، ﻣﻤﺎ ﻳﻌﻮﺩ ﺑﺎﻟﻨﻔﻊ ﻭﺍﻟﻔﺎﺋﺪﺓ ﻋﻠﻰ ﺇﻧﺴﺎﻥ ﺍﻟﺤﻲ، ﻓﻀﻼ‌ً ﻋﻦ ﺍﻟﺤﻲ ﻧﻔﺴﻪ ﻭﺍﻟﻤﺪﻳﻨﺔ ﻋﻠﻰ ﺣﺪ ﺳﻮﺍﺀ”.
  
 ﻭﺗﺎﺑﻊ ﻣﻌﺎﻟﻴﻪ “ﻭﻣﻦ ﻣﻈﺎﻫﺮ ﻧﺠﺎﺡ ﺍﻟﻔﻜﺮﺓ ﻣﺪﻯ ﺍﻟﺘﻌﺎﻭﻥ ﺍﻟﻘﺎﺋﻢ ﺑﻴﻦ ﺍﻟﺠﺎﻣﻌﺔ ﻭﺍﻟﻘﻄﺎﻋﺎﺕ ﺍﻟﺤﻜﻮﻣﻴﺔ ﺍﻟﺤﻴﻮﻳﺔ، ﻣﻨﻬﺎ: ﺃﻣﺎﻧﺔ ﺟﺪﺓ، ﺍﻟﺸﺮﻃﺔ، ﺍﻟﺪﻓﺎﻉ ﺍﻟﻤﺪﻧﻲ، ﻭﺯﺍﺭﺓ ﺍﻟﺼﺤﺔ، ﻓﻀﻼ‌ً ﻋﻦ ﻋﻤﺪ ﺍﻻ‌ﺣﻴﺎﺀ ﻭﻟﻬﻢ ﺟﻤﻴﻌﺎً ﺍﻟﺘﻘﺪﻳﺮ”.ﻭﺃﺿﺎﻑ ﻣﻌﺎﻟﻴﻪ “ﻣﻦ ﻫﺬﻩ ﺍﻟﻨﻈﺮﺓ ﺍﻟﻌﻤﻴﻘﺔ ، ﻧﺒﻌﺖ ﻓﻜﺮﺓ ﻣﺸﺮﻭﻉ “ﻛﻴﻒ ﻧﻜﻮﻥ ﻗﺪﻭﺓ” ﺿﻤﻦ ﻓﻌﺎﻟﻴﺎﺕ ﻣﻠﺘﻘﻰ ﻣﻜﺔ ﺍﻟﺜﻘﺎﻓﻲ ﻭﻫﻮ ﻣﻦ ﺑﻨﺎﺕ ﺃﻓﻜﺎﺭ ﺃﻣﻴﺮﻧﺎ ﺍﻟﻘﺪﻭﺓ ﺻﺎﺣﺐ ﺍﻟﺴﻤﻮ ﺍﻟﻤﻠﻜﻲ ﺍﻷ‌ﻣﻴﺮ ﺧﺎﻟﺪ ﺍﻟﻔﻴﺼﻞ ﻣﺴﺘﺸﺎﺭ ﺧﺎﺩﻡ ﺍﻟﺤﺮﻣﻴﻦ ﺍﻟﺸﺮﻳﻔﻴﻦ ﺃﻣﻴﺮ ﻣﻨﻄﻘﺔ ﻣﻜﺔ ﺍﻟﻤﻜﺮﻣﺔ ﺇﺫ ﺇﻥ ﻫﻤﻪ ﺍﻟﺸﺎﻏﻞ ﺍﻻ‌ﺭﺗﻘﺎﺀ ﺑﻤﻜﺔ ﺍﻟﻤﻜﺮﻣﺔ ﺣﺮﻣﺎً ﻭﺃﺭﺿﺎً ﻭﺇﻧﺴﺎﻧﺎً”.
  
 ﻭﻗﺪﻡ ﻣﻌﺎﻟﻴﻪ ﺷﻜﺮﻩ ﺍﻟﺠﺰﻳﻞ  ﻟﺼﺎﺣﺐ ﺍﻟﺴﻤﻮ ﺍﻟﻤﻠﻜﻲ ﺍﻷ‌ﻣﻴﺮ ﺧﺎﻟﺪ ﺍﻟﻔﻴﺼﻞ ﻣﺴﺘﺸﺎﺭ ﺧﺎﺩﻡ ﺍﻟﺤﺮﻣﻴﻦ ﺍﻟﺸﺮﻳﻔﻴﻦ ﺃﻣﻴﺮ ﻣﻨﻄﻘﺔ ﻣﻜﺔ ﺍﻟﻤﻜﺮﻣﺔ ﻭﻧﺎﺋﺐ ﺃﻣﻴﺮ ﻣﻨﻄﻘﺔ ﻣﻜﺔ ﺍﻟﻤﻜﺮﻣﺔ ﺻﺎﺣﺐ ﺍﻟﺴﻤﻮ ﺍﻟﻤﻠﻜﻲ ﺍﻷ‌ﻣﻴﺮ ﻋﺒﺪﺍﻟﻠﻪ ﺑﻦ ﺑﻨﺪﺭ ﻋﻠﻰ ﻫﺬﻩ ﺍﻟﻤﺒﺎﺩﺭﺓ ﺍﻟﻮﻃﻨﻴﺔ ﺍﻻ‌ﺟﺘﻤﺎﻋﻴﺔ ﻭﻋﻠﻰ ﺛﻘﺘﻪ ﺍﻟﻜﺮﻳﻤﺔ ﻓﻲ ﺟﺎﻣﻌﺔ ﺍﻟﻤﻠﻚ ﻋﺒﺪﺍﻟﻌﺰﻳﺰ ﻟﻺ‌ﺳﻬﺎﻡ ﻓﻲ ﺗﻨﻔﻴﺬ ﻫﺬﻩ ﺍﻟﻔﻌﺎﻟﻴﺎﺕ، ﻣﻦ ﺧﻼ‌ﻝ ﻣﺒﺎﺩﺭﺓ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﻭﺍﻟﺘﻲ ﺗﺴﺘﻬﺪﻑ ﺣﻲ ﺍﻟﺠﺎﻣﻌﺔ.
  
 ﻭﺑﻴﻦ ﻋﻤﻴﺪ ﺷﺆﻭﻥ ﺍﻟﻄﻼ‌ﺏ ﺍﻷ‌ﺳﺘﺎﺫ ﺍﻟﺪﻛﺘﻮﺭ ﻋﺒﺪﺍﻟﻤﻨﻌﻢ ﺑﻦ ﻋﺒﺪﺍﻟﺴﻼ‌ﻡ ﺍﻟﺤﻴﺎﻧﻲ ﺃﻥ  ﻣﺒﺎﺩﺭﺓ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﺗﻬﺪﻑ ﺇﻟﻰ ﺍﻟﻤﺴﺎﻫﻤﺔ ﻓﻲ ﺗﻄﻮﻳﺮ ﺍﻟﻤﻬﺎﺭﺍﺕ ﺍﻟﺸﺨﺼﻴﺔ ﻭﺗﻨﻤﻴﺔ ﺍﻟﺜﻘﺎﻓﺔ، ﺍﻟﻌﻤﻞ ﻋﻠﻰ ﺗﻘﺪﻳﻢ ﺍﻟﺨﻄﻂ ﻭﺍﻟﺪﺭﺍﺳﺎﺕ ﺍﻟﺘﻲ ﺗﺴﺎﻋﺪ ﻋﻠﻰ ﺍﻟﺮﻗﻲ ﺑﺎﻟﺤﻲ ﻭﺗﻄﻮﻳﺮ ﺧﺪﻣﺎﺗﻪ، ﻭﺍﻟﻤﺴﺎﻫﻤﺔ ﻓﻲ ﺗﻘﺪﻳﻢ ﺧﺪﻣﺎﺕ ﺍﻟﺒﻨﻴﺔ ﺍﻟﺘﺤﺘﻴﺔ ﻟﻠﺤﻲ ﻭﺍﻟﻌﻨﺎﻳﺔ ﺑﺎﺣﺘﻴﺎﺟﺎﺗﻪ ﺍﻷ‌ﺳﺎﺳﻴﺔ، ﻭﺍﻻ‌ﻫﺘﻤﺎﻡ ﺑﺎﻟﻤﺮﺍﻓﻖ ﺍﻟﻌﺎﻣﺔ “ﺍﻟﻤﺴﺎﺟﺪ، ﺍﻟﺤﺪﺍﺋﻖ، ﻭﺍﻟﺸﻮﺍﺭﻉ” ﻭﻇﻬﻮﺭﻫﺎ ﺑﺎﻟﺸﻜﻞ ﺍﻟﻼ‌ﺋﻖ ﻭﺍﻟﺠﺬﺍﺏ.
  
 ﻭﺗﻨﻔﺬ ﺍﻟﺠﺎﻣﻌﺔ 7 ﺑﺮﺍﻣﺞ ﺿﻤﻦ ﻣﺒﺎﺩﺭﺓ “ﺍﻟﺤﻲ ﺍﻟﻘﺪﻭﺓ” ﻭﻫﻲ:  “ﺍﻟﺤﻲ ﺍﻟﺘﻨﻤﻮﻱ” ﻭﺗﺮﻛﺰ ﻋﻠﻰ ﺇﻋﺎﺩﺓ ﺗﻬﻴﺌﺔ ﺍﻟﺒﻨﻴﺔ ﺍﻟﺘﺤﺘﻴﺔ ﻟﻠﺤﻲ، ﻭ”ﺍﻟﺤﻲ ﺍﻟﻤﺜﻘﻒ” ﻭﻫﻲ ﻋﺒﺎﺭﺓ ﻋﻦ ﺑﺮﺍﻣﺞ ﺗﺜﻘﻴﻔﻴﺔ ﻷ‌ﺑﻨﺎﺀ ﺍﻟﺤﻲ، ﻭ”ﺍﻟﺤﻲ ﺍﻟﻤﺘﻌﻠﻢ” ﻭﻳﺸﻤﻞ ﺑﺮﺍﻣﺞ ﺗﻌﻠﻴﻤﻴﺔ ﻷ‌ﺑﻨﺎﺀ ﺍﻟﺤﻲ، ﻭ”ﺍﻟﺤﻲ ﺍﻟﺼﺤﻲ” ﻭﻫﻮ ﺑﺮﺍﻣﺞ ﺗﻮﻋﻮﻳﺔ ﺻﺤﻴﺔ، ﻭ”ﺍﻟﺤﻲ ﺍﻟﻮﺍﻋﻲ” ﻭﻫﻮ ﺑﺮﺍﻣﺞ ﺗﻮﻋﻮﻳﺔ ﻣﺨﺘﻠﻔﺔ، ﻭ”ﺍﻟﺤﻲ ﺍﻵ‌ﻣﻦ” ﻭﻳﻘﺪﻡ ﺑﺮﺍﻣﺞ ﻣﺨﺘﻠﻔﺔ ﻓﻲ ﺍﻷ‌ﻣﻦ ﻭﺍﻟﺴﻼ‌ﻣﺔ، ﻭ”ﺍﻟﺤﻲ ﺍﻟﺠﻤﻴﻞ” ﻭﻳﻘﺪﻡ ﺑﺮﺍﻣﺞ ﻓﻲ ﻧﻈﺎﻓﺔ ﺍﻟﺤﻲ ﻭﺗﻨﻈﻴﻤﻪ.
  
  
  
  
  
  
  
  
  
  
  
  
  
  
 ترجع مشكلة الإمساك إلى عادات يومية سيئة، وبالتالي فإن تغيير هذه العادات وتبني عادات صحية يمكن أن يساهم كثيراً في التخلص من الإمساك:
 أصبح من المعروف اليوم أن على المرء أن يشرب يومياً ما بين لتر ونصف ولترين من السوائل من أجل صحة جيدة.
 هذه العادة تصبح أمراً ضرورياً في حال المعاناة من الإمساك، إذ أن شرب ما يكفي من السوائل يساعد على تطرية الفضلات وإفراغها.
 لذا اشرب الكثير من الماء وعصير الفاكهة والشوربات دون المبالغة في شرب القهوة والشاي.
 2. اشرب عصير الفاكهة الحمضية على الريق:
 يُعتبر عصير الحمضيات (البرتقال والكريفون…) مليناً طبيعياً ممتازاً.
 كما أن تأثير هذه العصائر على حركة الأمعاء يصبح أكثر فعالية فيما لو عوّدت نفسك على شربها طازجة وعلى الريق يومياً، مما يساعدك على التخلص من الإمساك ويزودك بالفيتامينات لتبدأ يومك بحيوية ونشاط.
 أصبح طعامنا اليوم، وللأسف، فقيراً بالألياف الطبيعية بسبب كثرة اللجوء إلى الوجبات السريعة والسوقية.
 علماً بأن الألياف تساعد الأمعاء في التخلص من فضلات جسمك، لذا تناول المزيد من الفاكهة والخضار الطازجة والبقول.
 واختار من الحبوب ما هو كامل ولم تُنزع قشرته (الأرز البني والمعكرونة السمراء والخبز من القمح الكامل…)، فالحبوب المكررة قد خسرت أليافها كما خسرت قيمتها الغذائية.
 وصفة الجدّات قد سبق أن أثبتت فعاليتها: بضعة مشمشات مجففة أو خوخات مجففة أو تينات مجففة أو تمرات على الريق أفضل علاج طبيعي للإمساك.
 تساهم الحياة التي تفتقر إلى الحركة في المعاناة من الإمساك. إذ أن تقوية عضلات البطن بممارسة منتظمة لتمارين رياضة يساعد على التخلص من الفضلات الراكدة في الأمعاء.
 تساهم الشدة النفسية وقلة الوقت المخصص للحمام في المعاناة من الإمساك.
 إن إفراغ الفضلات يتطلب من الجسم استدعاء عضلات مختلفة، وهذه العضلات بحاجة لتقوم بعملها على أحسن وجه أن تكون مسترخياً.
 لذا اجلس على مقعد الحمام على نحو مريح لتساعد نفسك على الاسترخاء.
 حدد موعداً يومياً للذهاب إلى الحمام، بهذه الطريقة تدرب أمعاءك وعضلاتك على القيام بهذا العمل بانتظام.
 ومرة أخرى خذ وقتك ولا تستعجل الأمور، إن أصريت وتابعت سيعتاد جسمك في النهاية، فالعادات الجيدة تحتاج أحياناً إلى وقت طويل من التمرن.
 قد يضطر بعض الأشخاص لأسباب عدة عدم تلبية متطلبات جسمهم، عليك ألا تتدخل وتغير أوقات ذهابك إلى الحمام لأن هذا الأمر قد يدفع بجسمك إلى عدم المطالبة ثانية وبالتالي ستعاني من الإمساك.
