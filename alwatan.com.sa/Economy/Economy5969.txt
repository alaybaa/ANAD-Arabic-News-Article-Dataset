انطلاق V20 لتوفير حلول مبنية على القيم ورفعها لقادة الدول 

انطلقت مجموعة القيم ( V20 ) التي تُعنى بالقيم كمحفز أساسي للسلوك، لأول مرة كمجموعة غير رسمية على ھامش أعمال مجموعة دول العشرين، وذلك بالتزامن مع رئاسة المملكة العربية السعودية لقمة دول مجموعة العشرين. وتـأتـي أھـّميّة المجـموعـة، لـما لـلقيم مـن أثـر عـميق يـُمّكن مـن الـتحول المسـتدام ورفـع جـودة الـحياة، حـيث يـسھم الـتعاطـي مـع الـمشكلات الـعالـمية مـن مـنظور القيم في معالجة جذور وأسباب التحديات بدلاً من ظواھرھا. وسـعياً مـن أعـضاء مجـموعـة الـتواصـل (V20) إلـى إبـراز الأثـر الإيـجابـي الـذي يـمكن حـدوثـه عـند إيـجاد حـلول مـبنية عـلى الـقيم، اسـتفادت المجـموعـة مـن الإمـكانـات الـھائـلة لـلمتخصصين والـممارسـين الـمشاركـين بـھا، وعـملت عـلى إنـشاء مـجتمع لـتبادل الـمعرفـة والابـتكار فـي وضـع السـياسـات ذات الأثـر الـملموس، وذلـك مـن مـنطلق أن تـمكين الـمعارف والأبـحاث فـي ھـذا الـمجال ستحسن من توجيه عملية صياغة السياسات الإقليمية والوطنية والعالمية، وقـد شـكلت المجـموعـة ثـلاث فـرق عـمل تـضمنت: فـريـق "الـقيم الـعالـمية" الــذي يــعمل عــلى إسهام الــقيم فــي تــحقيق الأھــداف الاجــتماعــية والاقــتصاديــة، وفــريــق "الــقيم فــي الــقيادة" الــذي يــعمل عــلى تحســين المخـرجـات الـتنظيمية، وفـريـق "الـقيم لـجودة الـحياة" الـذي يـركـز عـلى كـيفية تعزيز الرفاه على مستوى الأفراد والجماعات.وضـع مـلخصات السـياسـات ومـن الـمنتظر أن تـعمل الـفرق جـميعھا عـلى الـتشاور، وإجـراء الـبحوث، ووضـع مـلخصات السـياسـات، وتـقديـم الـنتائـج الـتي تـوصـلت إلـيھا فـي قـمة مجـموعـة تـواصـل الـقيم )V20( الـسنويـة، الـتي سـتعقد خـلال يـومـي 9-10مـن شھر نوفمبر القادم. وتـعمل المجـموعـة عـلى إطـلاق حـملة عـالـمية لـلمشاركـة الـمجتمعية بـدًءا مـن 15 أكـتوبـر الـمقبل، الـذي يـوافـق يـوم الـقيم الـعالـمي، وتھـدف إلـى تـشجيع الأفـراد فـي كـل أنـحاء الـعالـم عـلى مـشاركـة الأفـكار المجـديـة لـتعزيـز الـقيم فـي حـياتـھم الـفرديـة ومـجتمعاتـھم ودولـھم والـعالـم أجـمع، وسـتُتاح الأفـكار التي ُجمعت حول القيم العملية للعامة على منصة لتبادل المعرفة. وقال رئــيس مجــموعــة تــواصــل الــقيم (V20) الـمديـر الـتنفيذي لـمسك الـقيم قـائـلة ديــمة آل الشــيخ: "تـماشـيًا مـع مـبادئ الـتعاون الـدولـي لمجـموعـة العشـريـن الـتي تـحتفي بـالاخـتلافـات الـعالـمية، وسـد الـفجوة الـناتـجة عـنها وسـعيًا لـتحقيق الأهـداف الإنـسانـية المشـتركـة، يـسعدنـا إطـلاق مجـموعـة تـواصـل الـقيم غـير الـرسـمية (V20) خـلال الـعام الـذي تـترأس فـيه الـمملكة الـعربـية الـسعوديـة مجـموعـة العشـريـن، وسـتعكس المجـموعـة الـولـيدة الـنهج الـتعاونـي الـتصاعـدي الـذي تـتبعه مجـموعـات الـتواصـل الأخـرى تـحت مـظلة مجـموعـة العشـريـن لتيسـير الـمناقـشات وتـقديـم الـتوصـيات إلـى الـقادة، سـاعـين إلـى أن يسـتمر عـمل المجـموعـة خـلال الأعـوام الـقادمـة بـما يـضمن الوصول للتأثير المنشود".التحـديـات الـعالـمية أوضحت أن أعـضاء مجـموعـة تـواصـل الـقيم (V20) يؤمنون بـالـدور الـمحوري لـلقيم فـي تـشكيل سـلوك الـناس وتحسـين الـنتائـج الـمجتمعية، بـل ويـتعدى ذلـك إلـى الـمساعـدة فـي الـتغلب عـلى التحـديـات الـعالـمية، مبينة أنه مـن خـلال الجـمع بـين الـمفكريـن والـمنفذيـن، تـوضـح ھـذه المجـموعـة أھـمية الـقيم لـ ُصناع الـقرار فـي جميع القطاعات، إلى جانب أهميتها للأفراد بالقدر نفسه، مفيدة أن مجـموعـة تـواصـل الـقيم (V20) تبين أن السـياسـات الـقائـمة عـلى الـقيم يـمكن أن تـحسن حـياة الأفـراد فـي جـميع أنـحاء الـعالـم، وتـتيح المجـموعـة الـتواصـل مـع الـعامـلين فـي مـجال الـمبادرات الـقيمية حـول الـعالـم عـبر منصتين لمشاركة أفضل التجارب والطرق المتبعة لديهم. من جانبه، قال الـعضو الـمنتدب لأكـاديـمية بـاريـت للنهوض بـالـقيم الإنـسانـية ومـؤسـس مـركـز بـاريـت لـلقيم ريـتشارد بـاريـت : " يـسعدنـا فـي مـركـز بـاريـت لـلقيم أن نـشارك فـي مجـموعـة تـواصـل الـقيم (V20)، كون مـا نـقوم بـه فـي الـمركـز يُـثري الـفراغ الـمعرفـي فـي فهم كـيفية ضـمان تـوجـيه الـقيم للسـياسـات الـعامـة، ولهذا السـبب طـورت مجـموعـة تـواصـل الـقيم آلـيات فـّعالـة لإشـراك أكـبر عـدد مـمكن مـن الـخبراء والـمنظمات مـن خـلال تـكويـن ثـلاثـة فـرق عـمل ومـجتمع مـن الـمتخصصين، وسـتحصر الحـملة لمجـموعـة تـواصـل الـقيم (V20) أفـضل الـممارسـات الـمتعلقة بـالأفـكار المجـديـة عـلى أرض الـواقـع فـي السـياسـات ومـختلف الأنشـطة الـتي تـتمحور حـول الـقيم، كـما سـتساعـد المجـموعـة فـي اكـتشاف الـقيم الـعملية واسـتعراضها والـمساعـدة فـي تحقيقها". وسـتسهم كـل مـن الشـبكة، وحـملة الـمشاركـة الـمجتمعية، وجـهود فـرق الـعمل فـي إصـدار الـبيان الـختامـي لمجـموعـة تـواصـل الـقيم (V20) هذا العام، مما يشكل خطوة سباقة لهذه المجموعة الناشئة.مجـموعـات تـعاونـية مسـتقلة مجـموعـة تـواصـل الـقيم (V20) ھـي مجـموعـة ولـيدة تـأسسـت عـام 2020 عـلى أمـل أن تـتبناهـا مجـموعـة العشـريـن مـع مـرور الـوقـت لـتصبح ضـمن مجـموعـات الـتواصـل الـرسـمية، وھـي مجـموعـات تـعاونـية مسـتقلة تـقودها مـنظمات الـمجتمع الـمدنـي فـي الـبلد الـمضيف لـلقمة، وتـتعاون مجـموعـات الـتواصـل هذه مـع مجـموعـة واسـعة مـن أصـحاب الـمصلحة والـجهات الـمعنية فـي أنـحاء الـعالـم؛ لإجـراء الـبحوث والـمناقـشات والاتـصالات مـن أجـل وضـع تـوصـيات للسـياسـات الـعامـة لـينظر فيها قـادة مجموعة العشرين في القمة السنوية. ومـن الـمنتظر أن يجـري الـعمل عـلى بـناء الـقدرات الـتنظيمية لـمجـموعـة تــواصــل الــقيم (V20) طــوال عــام 2020؛ لتسهيل الاســتمراريــة فــي المســتقبل، وســتستضيفھا جــھة تــنظيمية مــختلفة كــل عــام لــضمان انــتقال التسھـيلات المسـتمرة والـمعرفـة الـمؤسسـية مـن أجـل دعـم الـتواصـل وإدارتـھا وتشجيع المزيد من النمو.
